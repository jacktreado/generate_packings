160 Quaternion Quaternion::operator% (const Quaternion &qright){
161  Quaternion qout;
162  double vecps,vecpx,vecpy,vecpz;
163  double crosspx,crosspy,crosspz;
164  double q1s,q1x,q1y,q1z;
165  double q2s,q2x,q2y,q2z;
166 
167  // get s,x,y,z for qleft
168  q1s = s;
169  q1x = x;
170  q1y = y;
171  q1z = z;
172 
173  // get s,x,y,z for qright
174  q2s = qright.get_s();
175  q2x = qright.get_x();
176  q2y = qright.get_y();
177  q2z = qright.get_z();
178 
179  // get cross product components
180  crosspx = q1y*q2z - q1z*q2y;
181  crosspy = -(q1x*q2z - q1z*q2x);
182  crosspz = q1x*q2y - q1y*q2x;
183 
184  vecps = q1s*q2s - (q1x*q2x + q1y*q2y + q1z*q2z);
185  vecpx = q1s*q2x + q2s*q1x + crosspx;
186  vecpy = q1s*q2y + q2s*q1y + crosspy;
187  vecpz = q1s*q2z + q2s*q1z + crosspz;
188 
189  // set values of output quaternion
190  qout.set_s(vecps);
191  qout.set_x(vecpx);
192  qout.set_y(vecpy);
193  qout.set_z(vecpz);
194 
195  return qout;
