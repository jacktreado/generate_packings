37 void packing::mxwl_vel_init (double t)
38 /* This subroutine assigns velocities according to the Maxwell
39  distribution. N is the total number of velocity components and
40  M is the total number of degrees of freedom. T is the system
41  temperature in the reduced units. Copyright (c) Tao Pang 1997. */
42 {
43  int i,d;
44  double v1;
45  double ek,vs;
46 
47  void grnf(double&);
48 
49  /* seed random number generator */
50  srandom((unsigned int) seed);
51 
52  /* Assign a Gaussian distribution to each velocity component */
53  for (i=0; i<N; i++){
54  for (d=0; d<NDIM; d++){
55  grnf(v1);
56  v[i][d] = v1;
57  }
58  }
59 
60  /* Scale the velocity to satisfy the equipartition theorem */
61  ek = 0;
62  for (i=0; i<N; i++){
63  for (d=0; d<NDIM; d++)
64  ek += 0.5*m[i]*v[i][d]*v[i][d];
65  }
66  ek /= N;
67 
68  vs = sqrt(ek/t);
69  for (i=0; i<N; i++){
70  for (d=0; d<NDIM; d++)
71  v[i][d] /= vs;
72  }
73 }

75 void grnf (double& x)
76 /* Two Gaussian random numbers generated from two uniform
77  random numbers. Copyright (c) Tao Pang 1997. */
78 {
79  double pi,r1,r2,r;
80 
81  pi = 4*atan(1.0);
82  r = (double)random()/(RAND_MAX+1.0);
83  r1 = -log(1-r);
84  r = (double)random()/(RAND_MAX+1.0);
85  r2 = 2*pi*r;
86  r1 = sqrt(2*r1);
87  x = r1*cos(r2);
88 }

91  int i,d;
92  double ek,vs;
93  double* pmean;
94  srand48(seed);
95 
96  pmean = new double[NDIM];
97  for (d=0; d<NDIM; d++)
98  pmean[d] = 0.0;
99 
100  for (i=0; i<N; i++){
101  for (d=0; d<NDIM; d++){
102  v[i][d] = drand48();
103  pmean[d] += m[i]*v[i][d];

105  }
106 
107  for (d=0; d<NDIM; d++)
108  pmean[d] /= N;
109

222 void packing::hs_force(){
223  int i,j,d,cind;
224  double sij, dx;
225  double xij[NDIM];
226 
227  // reset U
228  U = 0;
229 
230  for (i=0; i<N; i++){
231  for (j=i+1; j<N; j++){
232  // contact matrix index
233  cind = N*i + j - ((i+1)*(i+2))/2;
234 
235  // get contact distance sij
236  sij = r[j] + r[i];
237 
238  // get distance between particles
239  dx = this->get_distance(i,j,xij);
240 
241  // particles in contact, calculate force
242  if (dx < sij){
243  for (d=0; d<NDIM; d++){
244  F[i][d] += (this->hs(sij,dx))*xij[d];
245  F[j][d] -= (this->hs(sij,dx))*xij[d];
246  }
247 
248  // update contact matrix
249  c[cind] = 1;
250 
251  // update contact list
252  pc[i]++;
253  pc[j]++;
254 
255  // update potential energy
256  U += (ep/2)*pow(1-dx/sij,2);
257  }
258  // else, no force added and no contact
259  else
260  c[cind] = 0;
261  }
262  }

349 int packing::rmv_rattlers(int& krcrs) {
350  int i,j,ci,cj,r,nr,nm;
351 
352  // monitor recursion depth
353  krcrs++;
354 
355  // number of rattlers
356  nr = 0;
357 
358  // number of "marginal" rattlers to be removed
359  nm = 0;
360 
361  // loop over rows, eliminate contacts to rattlers
362  for (i=0; i<N; i++){
363  // start with r=0 contacts
364  r = 0;
365 
366  // count contacts
367  for (j=0; j<N; j++){
368  if (j < i){
369  ci = N*j + i - ((j+1)*(j+2))/2;
370  r += c[ci];
371  }
372  else if (j > i){
373  cj = N*i + j - ((i+1)*(i+2))/2; // mapping from matrix space to sub matrix space
374  r += c[cj];
375  }
376  else
377  continue;
378  }
379 
380  // remove from network if r <= DOF, delete contacts
381  if (r <= DOF){
382  // increment # of rattlers
383  nr++;
384 
385  // alter pc
386  pc[i] = 0;
387 
388  // if in contact, remove contacts
389  if (r > 0){
390  nm++;
391  for (j=0; j<N; j++){
392  if (j < i){
393  ci = N*j + i - ((j+1)*(j+2))/2;
394  if (c[ci] > 0){
395  c[ci] = 0;
396  pc[j]--;
397  }
398  }
399  else if (j > i){
400  cj = N*i + j - ((i+1)*(i+2))/2; // mapping from matrix space to sub matrix space
401  if (c[cj] > 0){
402  c[cj] = 0;
403  pc[j]--;
404  }
405  }
406  else
407  continue;
408  }
409  }
410  }
411  }
412

414  cout << "max recursive depth reached, be wary of output" << endl;
415  return -1;
416  }
417

419  if (nm == 0)

670  /*
671  GROWTH ALGORITHM
672 
673  1. If U < Utol grow
674  2. If U > 2*Utol & K < Ktol, stop, shrink by dphi0/2
675  a. If U < Utol & K < Ktol, relaxed, so grow again
676  b. If U > 2*Utol & K < Ktol, still overcompressed, so shrink by dphi0/2
677  3. Jammed when K < Ktol & Utol < U < 2Utol
678  */
679 
680  int nbb,niso,csum;
681  bool gr,oc,uc,marginal,jammed;
682  double dphi = dphi0;
683 
684  gr = 0;
685  oc = 0;
686  uc = 0;
687  marginal = 0;
688  jammed = 0;
689

692  csum = this->get_c_sum();
693 
694  gr = (U < Utol);
695  oc = (U > Utol && K < Ktol && csum >= niso && epconst == 1);
696  uc = (U < Utol);

698  jammed = (U > Utol && U < 2*Utol && K < Ktol && csum == niso && nr < N);
